81  /**
82  * Get the ratio of the explicit height of this Drawing to the implicit height of its contents.
83  *
84  * @return The ratio of the explicit height of this Drawing to the implicit height of its contents
85  */
86  private SisuBigDecimal getExplicitHeightPerImplicitHeight()
87  {
88  SisuBigDecimal implicitHeightOfContents = this.getImplicitHeightOfContents();
89  SisuBigDecimal explicitHeightPerImplicitHeight;
90  if (implicitHeightOfContents.isNotEqualTo(SisuBigDecimal.ZERO))
91  {
92  if (this.getExplicitHeight() == null)
93  {
94  // If the explicit height has no value, try to calculate using the explicit width
95  if (this.getExplicitWidth() == null)
96  {
97  // If the explicit width also has no value, the ratio is 0
98  explicitHeightPerImplicitHeight = SisuBigDecimal.ZERO;
99  }
100  else
101  {
102  // The explicit width exceeds 0, so use it to calculate the ratio
103  explicitHeightPerImplicitHeight = this.getExplicitWidth().divide(this.getImplicitWidthOfContents(),
104  SisuBigDecimal.mcOperations);
105  }
106  }
107  else
108  {
109  // The explicit height has a value, so calculate using the explicit height
110  explicitHeightPerImplicitHeight = this.getExplicitHeight().divide(implicitHeightOfContents,
111  SisuBigDecimal.mcOperations);
112  }
113  }
114  else
115  {
116  // The implicit height of the contents is 0, so the ratio is 0
117  explicitHeightPerImplicitHeight = SisuBigDecimal.ZERO;
118  }
119  return explicitHeightPerImplicitHeight;
120  }

451  /**
452  * Set the explicit dimensions of this Drawing.
453  * <p>
454  * It is preferable to do it this way rather than setting width and height separately.
455  *
456  * @param explicitWidthOfDrawingFloat The explicit width of the Drawing.
457  * @param explicitHeightOfDrawingFloat The explicit height of the Drawing.
458  */
459  public void setExplicitDimensions(Float explicitWidthOfDrawingFloat, Float explicitHeightOfDrawingFloat)
460  {
461  logger.atFine().log("Entering setExplicitDimensions");
462  // Calculate implicit aspect ratio
463  SisuBigDecimal implicitHeightOfContents = this.getImplicitHeightOfContents();
464  SisuBigDecimal implicitWidthOfContents = this.getImplicitWidthOfContents();
465  SisuBigDecimal implicitAspectRatio = SisuBigDecimal.ZERO;
466  if (implicitHeightOfContents.isGreaterThan(SisuBigDecimal.ZERO))
467  {
468  implicitAspectRatio = implicitWidthOfContents.divide(implicitHeightOfContents,
469  SisuBigDecimal.mcOperations);
470  }
471 
472  // Calculate explicit aspect ratio
473  this.setExplicitHeightInternal(SisuBigDecimal.valueOf(explicitWidthOfDrawingFloat));
474  this.setExplicitWidthInternal(SisuBigDecimal.valueOf(explicitWidthOfDrawingFloat));
475  SisuBigDecimal explicitAspectRatio = this.getExplicitWidth().divide(this.getExplicitHeight(),
476  SisuBigDecimal.mcOperations);
477  logger.atFine().log("Explicit aspect ratio: " + explicitAspectRatio);
478 
479  if (FALSE)
480  {
481  if (implicitAspectRatio.isGreaterThan(explicitAspectRatio))
482  {
483  // The implicit aspect ratio is greater than the explicit aspect ratio.
484  // Therefore, we are constrained by width.
485  // Adjust the height to match.
486  logger.atFine().log("Adjusting height");
487  SisuBigDecimal adjustedHeight = this.getExplicitWidth().divide(explicitAspectRatio,
488  SisuBigDecimal.mcOperations);
489  logger.atFine().log("Setting adjusted height, adjusted height: " + adjustedHeight.toPlainString());
490  this.setExplicitHeight(adjustedHeight);
491  logger.atFine().log("Setting explicit width");
492  this.setExplicitWidth(SisuBigDecimal.valueOf(explicitWidthOfDrawingFloat));
493  }
494  else if (implicitAspectRatio.isLessThan(explicitAspectRatio))
495  {
496  // The implicit aspect ratio is less than or equal to the explicit aspect ratio.
497  // Therefore, we are constrained by height.
498  // Adjust the width to match.
499  logger.atFine().log("Adjusting width");
500  SisuBigDecimal adjustedWidth = this.getExplicitHeight().multiply(explicitAspectRatio,
501  SisuBigDecimal.mcOperations);
502  this.setExplicitHeight(SisuBigDecimal.valueOf(explicitHeightOfDrawingFloat));
503  this.setExplicitWidth(adjustedWidth);
504  }
505  }
506  else
507  {
508  // The two aspect ratios are equal, so no adjustment is necessary
509  logger.atFine().log("No adjustment necessary");
510  this.setExplicitWidth(SisuBigDecimal.valueOf(explicitWidthOfDrawingFloat));
511  this.setExplicitHeight(SisuBigDecimal.valueOf(explicitHeightOfDrawingFloat));
512 
513  }
514 
515  }

680  private void updateExplicitXPositionOfShape(@NotNull Shape shape)
681  {
682  // Update the explicit x position of the Shape
683  logger.atFine().log("Updating explicit x position");
684  SisuBigDecimal implicitXPositionOfShape = shape.getImplicitXPositionCenter();
685  // The fudge factor is to shift the diagram right so that all x coordinates are positive in explicit coordinate
686  // space
687  SisuBigDecimal fudgeFactorX = this.getImplicitXMinimum();
688  SisuBigDecimal fudgedImplicitXPositionOfShape = implicitXPositionOfShape.subtract(fudgeFactorX,
689  SisuBigDecimal.mcOperations);
690  SisuBigDecimal explicitXPositionOfShape = fudgedImplicitXPositionOfShape.multiply(this.getExplicitToImplicitRatio(),
691  SisuBigDecimal.mcOperations);
692  shape.setExplicitXPositionCenter(explicitXPositionOfShape);
693  SisuBigDecimal implicitWidthOfContents = this.getImplicitWidth();
694  SisuBigDecimal explicitWidthOfContents = implicitWidthOfContents.multiply(this.getExplicitToImplicitRatio(),
