301  /**
302  * Test that, when a Drawing has three adjacent Shapes, then their x positions are correct.
303  *
304  * This method centralizes the test code.
305  *
306  * @param widthFloat The width of the drawing to be tested
307  * @param heightFloat The height of the drawing to be tested
308  * @param softly AssertJ behavior-driven development soft assertions
309  */
310  void test3Shapes(Float widthFloat, Float heightFloat, @NotNull BDDSoftAssertions softly)
311  {
312  drawing.add(shape1);
313  drawing.add(shape2);
314  drawing.add(shape3);
315  shape2.setRightOf(shape1);
316  shape3.setRightOf(shape2);
317  logger.atFine().log("width: " + widthFloat);
318  logger.atFine().log("height: " + heightFloat);
319  drawing.setExplicitWidth(widthFloat);
320  softly.then(drawing.getExplicitWidth().equals(SisuBigDecimal.valueOf(widthFloat)));
321  drawing.setExplicitHeight(heightFloat);
322  softly.then(drawing.getExplicitHeight().equals(SisuBigDecimal.valueOf(heightFloat)));
323 
324  // At this aspect ratio, the sizes of the shapes may be constrained by the height
325  SisuBigDecimal heightSisuBigDecimal = SisuBigDecimal.valueOf(heightFloat);
326  SisuBigDecimal widthSisuBigDecimal = SisuBigDecimal.valueOf(widthFloat);
327 
328  SisuBigDecimal aspectRatio = SisuBigDecimal.valueOf(widthFloat/heightFloat);
329 
330  SisuBigDecimal widthOfShapes;
331  if (aspectRatio.isGreaterThan(3))
332  {
333  widthOfShapes = heightSisuBigDecimal.multiply(SisuBigDecimal.valueOf(3));
334  }
335  else
336  {
337  widthOfShapes = widthSisuBigDecimal;
338  }
339 
340  SisuBigDecimal extraWidth = widthSisuBigDecimal.subtract(widthOfShapes);
341  SisuBigDecimal extraWidthLeft = extraWidth.divide(SisuBigDecimal.TWO, SisuBigDecimal.mcOperations);
342 
343  SisuBigDecimal circle1ExpectedXPosition = widthOfShapes.divide(SisuBigDecimal.valueOf(6), SisuBigDecimal.mcOperations);
344  if (aspectRatio.isGreaterThan(3))
345  {
346  circle1ExpectedXPosition = circle1ExpectedXPosition.add(extraWidthLeft);
347  }
348 
349  SisuBigDecimal circle1ExplicitXPosition = shape1.getExplicitXPositionCenter();
350  softly.then(circle1ExplicitXPosition.compareToFuzzy(circle1ExpectedXPosition))
351  .as("Expecting the explicit x position of shape 1 to be " +
352  circle1ExpectedXPosition + " but it was " + circle1ExplicitXPosition).isEqualTo(0);
353 
354  SisuBigDecimal circle2ExpectedXPosition = widthOfShapes.divide(SisuBigDecimal.TWO, SisuBigDecimal.mcOperations);
355  if (aspectRatio.isGreaterThan(3))
356  {
357  circle2ExpectedXPosition = circle2ExpectedXPosition.add(extraWidthLeft);
358  }
359 
360  SisuBigDecimal circle2ExplicitXPosition = shape2.getExplicitXPositionCenter();
361  softly.then(circle2ExplicitXPosition.compareToFuzzy(circle2ExpectedXPosition))
362  .as("Expecting the explicit x position of shape 2 to be " +
363  circle2ExpectedXPosition + " but it was " + circle2ExplicitXPosition).isEqualTo(0);
364 
365  SisuBigDecimal fraction = SisuBigDecimal.valueOf(5).divide(SisuBigDecimal.valueOf(6), SisuBigDecimal.mcOperations);
366  SisuBigDecimal circle3ExpectedXPosition = widthOfShapes.multiply(fraction, SisuBigDecimal.mcOperations);
367  if (aspectRatio.isGreaterThan(3))
368  {
369  circle3ExpectedXPosition = circle3ExpectedXPosition.add(extraWidthLeft);
370  }
371  SisuBigDecimal circle3ExplicitXPosition = shape3.getExplicitXPositionCenter();
372  softly.then(circle3ExplicitXPosition.compareToFuzzy(circle3ExpectedXPosition))
373  .as("Expecting the explicit x position of shape 3 to be " +
374  circle3ExpectedXPosition + " but it was " + circle3ExplicitXPosition).isEqualTo(0);
375  }
